65 /*
66  See this:
67 http://demonstrations.wolfram.com/SteadyStateTwoDimensionalConvectionDiffusionEquation/
68 
69  Solve this:
70  (Pr*Re)^-1 * (d2T/dx2 + d2T/dy2) - u dT/dx -v dT/dy = 0
71 
72  Domain (0,1) times (0,2)
73  Dirichlet BC on left wall and right wall
74  Tleft = 0 , Tright = 1
75  homogeneous Neumann on top and bottom
76 
77  u = - sin(pi x) cos(pi y)
78  v = cos(pi x) sin(pi y)
79 
80  Use 2nd order FD, with following grid:
81 
82  o x x x x o
83  o x x x x o
84  o x x x x o
85  o x x x x o
86 
87  The x denotes an unknown. o : known values
88  Enumeration of unknowns is done left-right from bottom to top
89  as follows:
90 
91  ...
92  o x x x x o
93 
94  4 5 6 7
95  o x x x x o
96 
97  0 1 2 3
98  o x x x x o
99 
100  MPI decomp is done (simply) by splitting blocks of the domains
101  vertically, and ranks go from 0 to n_ranks-1 starting from bottom.
102  */

160  void assembleMatrix() const
161  {
162  int gi{}; int gj{};
163  int k{};
164  std::array<int, 5> colInd{};
165  std::array<scalar_type, 5> values{};
166  scalar_type c_T_ip1{};
167  scalar_type c_T_im1{};
168  scalar_type c_T_ij {};
169  scalar_type c_T_jp1{};
170  scalar_type c_T_jm1{};
171  int gid_ip1{};
172  int gid_im1{};
173  int gid_jp1{};
174  int gid_jm1{};
175 
176  for (auto i=0; i<dofPerProc_; i++)
177  {
178  // global ID of current point
179  auto GID = MyGlobalDof_[i];
180  // find global i,j of this point
181  globalIDToGiGj(GID, gi, gj);
182 
183  const scalar_type uij = (*u_)[i];
184  const scalar_type vij = (*v_)[i];
185 
186  k = 0;
187  c_T_ij = - invPrRe_*dxSqInv_ - invPrRe_*dySqInv_;
188  values[k] = 2.0*c_T_ij;
189  colInd[k] = GID;
190  k++;
191 
192  if (gi>=1){
193  c_T_im1 = invPrRe_*dxSqInv_ + uij*dx2Inv_;
194  gid_im1 = GID - 1;
195  values[k] = c_T_im1;
196  colInd[k] = gid_im1;
197  k++;
198  }
199 
200  if (gi<NxDof_-1){
201  c_T_ip1 = invPrRe_*dxSqInv_ - uij*dx2Inv_;
202  gid_ip1 = GID + 1;
203  values[k] = c_T_ip1;
204  colInd[k] = gid_ip1;
205  k++;
206  }
207 
208  if (gj>=1 and gj<NyDof_-1){
209  c_T_jm1 = invPrRe_*dySqInv_ + vij*dy2Inv_;
210  gid_jm1 = GID - NxDof_;
211  values[k] = c_T_jm1;
212  colInd[k] = gid_jm1;
213  k++;
214 
215  c_T_jp1 = invPrRe_*dySqInv_ - vij*dy2Inv_;
216  gid_jp1 = GID + NxDof_;
217  values[k] = c_T_jp1;
218  colInd[k] = gid_jp1;
219  k++;
220  }
221 
222  if (gj==0){
223  c_T_jp1 = 2.0*invPrRe_*dySqInv_;
224  gid_jp1 = GID + NxDof_;
225  values[k] = c_T_jp1;
226  colInd[k] = gid_jp1;
227  k++;
228  }
229 
230  if (gj==NyDof_-1){
231  c_T_jm1 = 2.0*invPrRe_*dySqInv_;
232  gid_jm1 = GID - NxDof_;
233  values[k] = c_T_jm1;
234  colInd[k] = gid_jm1;
235  k++;
236  }
237 
238  if (A_->Filled())
239  A_->ReplaceGlobalValues(GID, k,
240  values.data(), colInd.data());
241  else
242  A_->InsertGlobalValues(GID, k,
243  values.data(), colInd.data());
244  }//loop
245 
246  if(!A_->Filled())
247  A_->FillComplete();
248  }

250  void fillRhs() const{
251  f_->PutScalar( static_cast<scalar_type>(0) );
252 
253  int gi{}; int gj{};
254  for (auto i=0; i<dofPerProc_; i++)
255  {
256  // global ID of current point
257  auto GID = MyGlobalDof_[i];
258  // find global i,j of this point
259  globalIDToGiGj(GID, gi, gj);
260 
261  if (gi==0){
262  auto value = invPrRe_*dxSqInv_ + (*u_)[i]*dx2Inv_;
263  (*f_)[i] = -value * Tleft_;
264  }
265 
266  if (gi==NxDof_-1){
267  auto value = invPrRe_*dxSqInv_ - (*u_)[i]*dx2Inv_;
268  (*f_)[i] = -value * Tright_;
269  }
270  }//loop
271  }

273  void solve(){
274  Epetra_LinearProblem Problem(A_.get(), T_.get(), f_.get());
275  AztecOO Solver(Problem);
276  Solver.Iterate(500, solveTolerance_);
277  Solver.NumIters();
278  Solver.TrueResidual();
279  }

310 protected:
311  void setup(){
312  createMap();
313  dofPerProc_= myMap_->NumMyElements();
314  MyGlobalDof_ = myMap_->MyGlobalElements();
315 
316  x_ = std::make_shared<nativeVec>(*myMap_);
317  y_ = std::make_shared<nativeVec>(*myMap_);
318  u_ = std::make_shared<nativeVec>(*myMap_);
319  v_ = std::make_shared<nativeVec>(*myMap_);
320  int gi{}; int gj{};
321  for (auto i = 0; i<dofPerProc_; i++){
322  auto GID = MyGlobalDof_[i];
323  globalIDToGiGj(GID, gi, gj);
324  (*x_)[i] = dx_ + gi * dx_;
325  (*y_)[i] = gj * dy_;
326  auto xval = (*x_)[i];
327  auto yval = (*y_)[i];
328  (*u_)[i] = -std::sin(M_PI*xval) * std::cos(M_PI*yval);
329  (*v_)[i] = std::cos(M_PI*xval) * std::sin(M_PI*yval);
330  }
331 
332  A_ = std::make_shared<nativeMatrix>(Copy, *myMap_, maxNonZeroPerRow_);
333  T_ = std::make_shared<nativeVec>(*myMap_);
334  f_ = std::make_shared<nativeVec>(*myMap_);
335  }
