109 private:
110  void setup()
111  {
112  constexpr auto one = ::pressio::utils::constants<scalar_type>::one();
113  constexpr auto two = ::pressio::utils::constants<scalar_type>::two();
114  constexpr auto oneHalf = one/two;
115  dx_ = (xR_ - xL_)/static_cast<scalar_type>(Ncell_);
116  dxInv_ = one/dx_;
117 
118  xGrid_.resize(Ncell_);
119  for (int_t i=0; i<Ncell_; ++i)
120  xGrid_(i) = dx_*static_cast<scalar_type>(i) + dx_*oneHalf;
121 
122  // init condition
123  U_.resize(Ncell_);
124  U0_.resize(Ncell_);
125  for (int_t i=0; i<Ncell_; ++i){
126  U_(i) = one;
127  U0_(i) = one;
128  }
129  };
130 
131  void velocity_impl(const state_type & u,
132  const scalar_type & t,
133  velocity_type & f) const
134  {
135  constexpr auto one = ::pressio::utils::constants<scalar_type>::one();
136  constexpr auto two = ::pressio::utils::constants<scalar_type>::two();
137  constexpr auto oneHalf = one/two;
138 
139  const auto coeff = oneHalf * dxInv_;
140  f(0) = coeff*(mu_[0]*mu_[0] - u(0)*u(0)) + mu_[1] * std::exp(mu_[2]*xGrid_(0));
141  for (int_t i=1; i<Ncell_; ++i){
142  f(i) = coeff*(u(i-1)*u(i-1) - u(i)*u(i)) + mu_[1]*std::exp(mu_[2]*xGrid_(i));
143  }
144  }
145 
146  void jacobian_impl(const state_type & u,
147  const scalar_type & t,
148  jacobian_type & jac) const
149  {
150  jac(0,0) = -dxInv_*u(0);
151  for (int_t i=1; i<Ncell_; ++i){
152  jac(i, i) = -dxInv_ * u(i);
153  jac(i, i-1) = dxInv_ * u(i-1);
154  }
155  }
156 
157 private:
158  const scalar_type xL_ = 0.0;
159  const scalar_type xR_ = 100.0;
160  std::array<scalar_type, 3> mu_ = {{5., 0.02, 0.02}};
161 
162  std::size_t Ncell_;
163  scalar_type dx_;
164  scalar_type dxInv_;
165  state_type xGrid_;
166 
167  mutable state_type U_;
168  state_type U0_;
