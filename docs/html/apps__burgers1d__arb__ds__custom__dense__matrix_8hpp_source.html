54 template <typename scalar_t>
55 class DenseMatrix
56 {
57 public:
58  using value_type = scalar_t;
59  using size_type = std::size_t;
60  using index_type = size_type;
61 
62  using data_type = std::vector<std::vector<scalar_t>>;
63 public:
64  // DenseMatrix() = default;
65 
66  DenseMatrix(index_type nRows, index_type nCols)
67  : data_(nRows), numRows_{nRows}, numCols_{nCols}{
68  for (auto & it : data_) it.resize(nCols);
69  }
70 
71  void resize(index_type newRows, index_type newCols){
72  data_.resize(newRows);
73  for (auto & it : data_) it.resize(newCols);
74  }
75 
76  index_type extent(index_type k) const{
77  assert(k <= 1);
78  assert(data_.empty() == false);
79 
80  return (k==0) ? numRows_ : numCols_;
81  }
82 
83  value_type & operator()(index_type i, index_type j){
84  return data_[i][j];
85  }
86  value_type const & operator()(index_type i, index_type j) const{
87  return data_[i][j];
88  }
89 
90  data_type * data(){
91  return &data_;
92  }
93  data_type const * data() const{
94  return &data_;
95  }
96 
97 private:
98  data_type data_;
99  size_type numRows_ {};
100  size_type numCols_ {};
101 
102 };//end class
