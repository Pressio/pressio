54 template<
55  typename scalar_t,
56  typename fom_system_type,
57  typename lspg_state_type,
58  typename lspg_residual_type,
59  typename lspg_jacobian_type,
60  typename residual_policy_type,
61  typename jacobian_policy_type
62  >
63 class System
64 {
65  typename std::conditional<
66  ::pressio::ops::predicates::is_object_pybind<fom_system_type>::value,
67  const fom_system_type, const fom_system_type & >::type fomSystemObj_;
68 
69  const residual_policy_type & residualEvaluator_;
70  const jacobian_policy_type & jacobianEvaluator_;
71  mutable lspg_residual_type R_;
72  mutable lspg_jacobian_type J_;
73 
74 public:
75  // these need to be public because are detected by solver
76  using scalar_type = scalar_t;
77  using state_type = lspg_state_type;
78  using residual_type = lspg_residual_type;
79  using jacobian_type = lspg_jacobian_type;
80 
81 public:
82  System() = delete;
83  ~System() = default;
84 
85  System(const fom_system_type & fomSystemObj,
86  const residual_policy_type & resPolicyObj,
87  const jacobian_policy_type & jacPolicyObj)
88  : fomSystemObj_(fomSystemObj),
89  residualEvaluator_(resPolicyObj),
90  jacobianEvaluator_(jacPolicyObj),
91  R_(residualEvaluator_.create(fomSystemObj_)),
92  J_(jacobianEvaluator_.create(fomSystemObj_))
93  {}
94 
95 public:
96  lspg_residual_type createResidual() const
97  {
98  return R_;
99  }
100 
101  lspg_jacobian_type createJacobian() const
102  {
103  return J_;
104  }
105 
106  void residual(const lspg_state_type & romState,
107  lspg_residual_type & R,
108  ::pressio::Norm normKind,
109  scalar_type & nrmValue) const
110  {
111  residualEvaluator_.template compute(romState, R, fomSystemObj_, normKind, nrmValue);
112  }
113 
114  void jacobian(const lspg_state_type & romState, lspg_jacobian_type & J) const
115  {
116  jacobianEvaluator_.template compute(romState, J, fomSystemObj_);
117  }
118 
119  // the following are needed to interface with the optimizers
120  scalar_type operator()(const state_type & romState) const
121  {
122  scalar_type normR = {};
123  this->residual(romState, R_, ::pressio::Norm::L2, normR);
124  return normR*normR;
125  }
126 
127  void gradient( const state_type & romState, state_type & g) const
128  {
129  scalar_type normR = {};
130  this->residual(romState, R_, ::pressio::Norm::L2, normR);
131  this->jacobian(romState, J_);
132  constexpr auto beta = ::pressio::utils::constants<scalar_type>::zero();
133  constexpr auto alpha = ::pressio::utils::constants<scalar_type>::two();
134  ::pressio::ops::product(::pressio::transpose(), alpha, J_, R_, beta, g);
135  }
136 };//end class
