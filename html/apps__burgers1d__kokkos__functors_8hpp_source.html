57 template <
58  typename x_t,
59  typename u_t,
60  typename rhs_t,
61  typename sc_t
62  >
63 struct VelocityFunctor{
64  sc_t mu0_;
65  sc_t mu1_;
66  sc_t mu2_;
67  int Ncell_;
68  sc_t dxInv_;
69  x_t x_;
70  u_t u_;
71  rhs_t r_;
72 
73  VelocityFunctor(sc_t mu0, sc_t mu1, sc_t mu2,
74  int Ncell, sc_t dxInv,
75  x_t x, u_t y, rhs_t r)
76  : mu0_{mu0}, mu1_{mu1}, mu2_{mu2},
77  Ncell_{Ncell}, dxInv_{dxInv},
78  x_{x}, u_{y}, r_{r}{}
79 
80  KOKKOS_INLINE_FUNCTION
81  void operator() (const int & i) const {
82  if (i==0)
83  r_(i) = 0.5 * dxInv_ * (mu0_*mu0_ - u_(i)*u_(i));
84 
85  if (i>=1 and i<Ncell_)
86  r_(i) = 0.5 * dxInv_ * (u_(i-1)*u_(i-1) - u_(i)*u_(i));
87 
88  r_(i) += mu1_*std::exp(mu2_*x_(i));
89  }
90 };

93 template <
94  typename u_t,
95  typename crs_mat_t,
96  typename sc_t
97  >
98 struct JacobianFunctor{
99  int Ncell_;
100  sc_t dxInv_;
101  u_t u_;
102  crs_mat_t J_;
103 
104  JacobianFunctor(int Ncell,
105  sc_t dxInv,
106  u_t y,
107  crs_mat_t Jin)
108  : Ncell_{Ncell},
109  dxInv_{dxInv},
110  u_{y},
111  J_{Jin}{}
112 
113  KOKKOS_INLINE_FUNCTION
114  void operator() (const int & iRow) const
115  {
116  const int ncol = (iRow==0) ? 1 : 2;
117  int cols[2];
118  sc_t vals[2];
119 
120  if (iRow==0){
121  cols[0] = 0;
122  vals[0] = -dxInv_*u_(iRow);
123  }
124  else{
125  cols[0] = iRow-1;
126  vals[0] = dxInv_*u_(iRow-1);
127 
128  cols[1] = iRow;
129  vals[1] = -dxInv_*u_(iRow);
130  }
131 
132  J_.replaceValues(iRow, cols, ncol, vals, false, true);
133  }
134 };
