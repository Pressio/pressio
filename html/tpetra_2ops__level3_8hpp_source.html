55  * for tpetra:
56  *
57  * C = beta * C + alpha*op(A)*op(B)
58  *

61 /* -------------------------------------------------------------------
62  * specialize for op(A) = A^T and op(B) = B
63  *-------------------------------------------------------------------*/
64 template <
65  typename A_type, typename B_type, typename scalar_type, typename C_type
66  >
67 ::pressio::mpl::enable_if_t<
68  ::pressio::containers::predicates::is_multi_vector_wrapper_tpetra<A_type>::value and
69  ::pressio::containers::predicates::is_multi_vector_wrapper_tpetra<B_type>::value
70  /*and ::pressio::containers::predicates::is_dense_matrix_wrapper_eigen<C_type>::value*/
71  >
72 product(::pressio::transpose modeA,
73  ::pressio::nontranspose modeB,
74  const scalar_type alpha,
75  const A_type & A,
76  const B_type & B,
77  const scalar_type beta,
78  ::pressio::containers::MatrixSharedMemBase<C_type> & C)
79 {
80  static_assert(containers::predicates::are_scalar_compatible<A_type, B_type, C_type>::value,
81  "Types are not scalar compatible");
82 
83  const auto numVecsA = A.numVectors();
84  const auto numVecsB = B.numVectors();
85  assert( (std::size_t)A.extent(0) == (std::size_t)B.extent(0));
86  assert( (std::size_t)C.extent(0) == (std::size_t)numVecsA );
87  assert( (std::size_t)C.extent(1) == (std::size_t)numVecsB );
88  // using ord_t = typename ::pressio::containers::details::traits<A_type>::global_ordinal_t;
89 
90  // compute dot between every column of A with every col of B
91  for (std::size_t i=0; i<(std::size_t)numVecsA; i++)
92  {
93  // colI is a Teuchos::RCP<Vector<...>>
94  const auto colI = A.data()->getVector(i);
95  for (std::size_t j=0; j<(std::size_t)numVecsB; j++)
96  {
97  const auto colJ = B.data()->getVector(j);
98  C(i,j) = beta * C(i,j) + alpha * colI->dot(*colJ);
99  }
100  }
101 }

130 // /***********************************
131 // * special case A==B
132 // **********************************/
133 template <typename A_type, typename scalar_type, typename C_type>
134 ::pressio::mpl::enable_if_t<
135  ::pressio::containers::predicates::is_multi_vector_wrapper_tpetra<A_type>::value and
136  ::pressio::containers::predicates::is_dense_matrix_wrapper_eigen<C_type>::value
137  >
138 product(::pressio::transpose modeA,
139  ::pressio::nontranspose modeB,
140  const scalar_type alpha,
141  const A_type & A,
142  const scalar_type beta,
143  C_type & C)
144 {
145  static_assert(containers::predicates::are_scalar_compatible<A_type, C_type>::value,
146  "Types are not scalar compatible");
147 
148  // how many vectors are in A and mvB
149  const auto numVecsA = A.numVectors();
150  assert((std::size_t)C.extent(0) == (std::size_t)numVecsA);
151  assert((std::size_t)C.extent(1) == (std::size_t)numVecsA);
152 
153  // using ord_t = typename ::pressio::containers::details::traits<A_type>::global_ordinal_t;
154  scalar_type tmp = ::pressio::utils::constants<scalar_type>::zero();
155 
156  // A dot A = A^T*A, which yields a symmetric matrix
157  // only need to compute half and fill remaining entries accordingly
158  for (std::size_t i=0; i<(std::size_t)numVecsA; i++)
159  {
160  // colI is a Teuchos::RCP<Vector<...>>
161  auto colI = A.data()->getVector(i);
162  for (std::size_t j=i; j<(std::size_t)numVecsA; j++)
163  {
164  auto colJ = A.data()->getVector(j);
165  tmp = alpha*colI->dot(*colJ);
166  C(i,j) = beta*C(i,j) + tmp;
167  C(j,i) = beta*C(j,i) + tmp;
168  }
169  }
170 }

173 template <typename A_type, typename scalar_type, typename C_type>
174 ::pressio::mpl::enable_if_t<
175  ::pressio::containers::predicates::is_multi_vector_wrapper_tpetra<A_type>::value and
176  ::pressio::containers::predicates::is_dense_matrix_wrapper_kokkos<C_type>::value
177  >
178 product(::pressio::transpose modeA,
179  ::pressio::nontranspose modeB,
180  const scalar_type alpha,
181  const A_type & A,
182  const scalar_type beta,
183  C_type & C)
184 {
185  static_assert(containers::predicates::are_scalar_compatible<A_type, C_type>::value,
186  "Types are not scalar compatible");
187 
188  static_assert(std::is_same<
189  typename containers::details::traits<A_type>::device_t,
190  typename containers::details::traits<C_type>::device_t
191  >::value,
192  "Non-matching device types");
193 
194  using map_t = typename ::pressio::containers::details::traits<A_type>::data_map_t;
195  using tpetra_mv_t = typename ::pressio::containers::details::traits<A_type>::wrapped_t;
196  const auto indexBase = A.data()->getMap()->getIndexBase();
197  const auto comm = A.data()->getMap()->getComm();
198 
199  // C should be square matrix
200  assert( (std::size_t)C.extent(0) == (std::size_t)C.extent(1) );
201  const auto n = C.extent(0);
202  Teuchos::RCP<const map_t> replMap(new map_t(n, indexBase, comm, Tpetra::LocallyReplicated));
203  // create multivector that views the Kokkos matrix
204  tpetra_mv_t Cmv(replMap, *C.data());
205 
206  // do the operation
207  Cmv.multiply(Teuchos::ETransp::TRANS, Teuchos::ETransp::NO_TRANS, alpha, *A.data(), *A.data(), beta);
208 }
