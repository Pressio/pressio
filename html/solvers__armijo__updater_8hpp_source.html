54 template <typename sc_t, typename state_t, typename T>
55 class ArmijoUpdater : public T
56 {
57  state_t trialState_;
58  const sc_t beta_ = 0.0001;
59 
60 public:
61  ArmijoUpdater() = delete;
62 
63  template <typename sys_t, typename ...Args>
64  ArmijoUpdater(const sys_t & sys, const state_t & state, Args &&... args)
65  : T(sys, state, std::forward<Args>(args)...), trialState_(state){}
66 
67  void resetForNewCall(){
68  T::resetForNewCall();
69  }
70 
71  template<typename system_t>
72  void updateState(const system_t & system, state_t & state)
73  {
74  constexpr auto one = ::pressio::utils::constants<sc_t>::one();
75  auto alpha = static_cast<sc_t>(1);
76 
77  // https://people.maths.ox.ac.uk/hauser/hauser_lecture2.pdf
78 
79  // Armijo rule says to backtrack alpha until:
80  // f(x_k + alpha_l * p_k) - f(x_k) <= alpha_l * beta * dot(g_k, p_k)
81  //
82  // where
83  // k = the GN step
84  // l = indexes the armijo backtracking)
85  // p_k is the correction at GN k-th step
86  // g_k is the gradient at GN k-th step
87 
88  const auto & p_k = T::getCorrection();
89  const auto & g_k = T::getGradient();
90  const auto fx_k = T::residualNormCurrentCorrectionStep();
91  const auto gkDotpk = ::pressio::ops::dot(g_k, p_k);
92 
93 #ifdef PRESSIO_ENABLE_DEBUG_PRINT
94  ::pressio::utils::io::print_stdout(" backtracking: alpha =", alpha, "\n");
95 #endif
96  sc_t ftrial = {};
97  bool done = false;
98  while (not done)
99  {
100  // update : trialState = x_k + alpha*p_k
101  ::pressio::ops::do_update(trialState_, state, one, p_k, alpha);
102 
103  // compute rhs_l = alpha_l * beta * dot(g_k, p_k)
104  const auto rhs = alpha * beta_ * gkDotpk;
105 
106  // eval f(x_k + alpha_l * p_k)
107  T::residualNorm(system, trialState_, ftrial);
108 
109  // lhs = f(x_k + alpha_l * p_k) - f(x_k)
110  const auto lhs = ftrial - fx_k;
111 
112 #ifdef PRESSIO_ENABLE_DEBUG_PRINT
113  ::pressio::utils::io::print_stdout(" f(x_k+alpha*p_k) =", ftrial, "\n");
114  ::pressio::utils::io::print_stdout(" f(x_k+alpha*p_k)-f(x_k) =", lhs,
115  "; alpha*beta*g^T p_k =", rhs, "\n");
116 #endif
117 
118  if (lhs <= rhs) done = true;
119 
120  // exit when abs(fytrail-fy) < eps, leave eps = 1e-14 for now
121  // change later with some machine epsilon
122  if (std::abs(lhs) <= 1e-14) done = true;
123 
124  /* convectional way to backtrack:alpha_l+1 = 0.5 * alpha_l */
125  if (!done) alpha *= 0.5;
126  }//while
127 
128  // solution update: state = state + alpha*p_k
129  ::pressio::ops::do_update(state, one, p_k, alpha);
130  }
131 };
