106 time_discrete_jacobian(lspg_matrix_type & jphi, //jphi holds J * phi
107  const scalar_type & dt,
108  const decoder_jac_type & phi)
109 {
110  auto & jphiNat = *jphi.data();
111  auto jphiMU = jphiNat.mutable_unchecked();
112 
113  auto & phiNat = *phi.data();
114  const auto phiMU = phiNat.unchecked();
115 
116  // prefactor (f) multiplying f*dt*J*phi
117  const auto prefactor = dt * dtPrefactor<stepper_tag, scalar_type>::value;
118  const auto nRows = jphi.extent(0);
119  const auto nCols = jphi.extent(1);
120  for (std::size_t i=0; i<(std::size_t)nRows; ++i){
121  for (std::size_t j=0; j<(std::size_t)nCols; ++j){
122  jphiMU(i,j) = phiMU(i,j) + prefactor*jphiMU(i,j);
123  }
124  }

169 /* for trilinos data structures, support hyper-reduction too.
170  * When we have hyper-reduction and we need to calculate the
171  * time-discrete residual, yn (i.e. the current state) and
172  * ynm (i.e. the states at prev steps) might have
173  * different entries/maps than R (i.e. the spatial residual)
174  * so we need to update R only picking elements corresponding
175  * elements in yn and ynm.
176  * In other words, in the functions below we do NOT assume that
177  * yn and ynm have the same distribution/sizes of R but we DO assume
178  * that yn and ynm contain at least all the elements of R (and more).
179  * THis is because to evaluate spatial residual one typically
180  * needs neighbors values, so in the hyper-red scenario the
181  * states contain more DOFs than the spatial residual.
182  * for instance, imagine the i-th cell in a uniform grid with a
183  * centered 2nd-order FD operator.
184  * To compute the spatial residual at the i-th cell, one needs the
185  * cell value y_i, as well as the left, y_i-1, and right, y_i+1,
186  * values of the state. So for a single DOF of the residual, one
187  * needs 3 DOFs of the state. Extrapolating to all cells, then we
188  * clearly see that we need more dofs for the states than for the residual.
189  *
190  * Below, we account for this potential scenario and the code below
191  * obviously also works for the case where the vectors have same
192  * the same distributions.
193 */

200 template <
201  typename stepper_tag,
202  typename lspg_matrix_type,
203  typename scalar_type,
204  typename decoder_jac_type
205 >
206 ::pressio::mpl::enable_if_t<
207  containers::predicates::is_multi_vector_wrapper_epetra<lspg_matrix_type>::value and
208  containers::predicates::is_multi_vector_wrapper_epetra<decoder_jac_type>::value
209  >
210 time_discrete_jacobian(lspg_matrix_type & jphi, //jphi stands for J * phi
211  const scalar_type & dt,
212  const decoder_jac_type & phi){
213 
214  // integral type of the global indices
215  using GO_t = typename containers::details::traits<lspg_matrix_type>::global_ordinal_t;
216 
217  //get row map of phi
218  const auto & phi_map = phi.data()->Map();
219  // get my global elements
220  std::vector<GO_t> gIDphi( phi.extentLocal(0) );
221  phi_map.MyGlobalElements( gIDphi.data() );
222 
223  // get map of jphi
224  const auto & jphi_map = jphi.data()->Map();
225  // get global elements
226  std::vector<GO_t> gIDjphi( jphi.extentLocal(0) );
227  jphi_map.MyGlobalElements( gIDjphi.data() );
228 
229  // prefactor (f) multiplying f*dt*J*phi
230  constexpr auto prefactor = dtPrefactor<stepper_tag, scalar_type>::value;
231 
232  //loop over elements of jphi
233  for (auto i=0; i<jphi.extentLocal(0); i++){
234  // ask the phi map what is the local index corresponding
235  // to the global index we are handling
236  auto lid = phi_map.LID(gIDjphi[i]);
237  for (auto j=0; j<jphi.numVectors(); j++)
238  jphi(i,j) = phi(lid,j) + prefactor*dt*jphi(i,j);
239  }
240 }

246 template <
247  typename stepper_tag,
248  typename lspg_matrix_type,
249  typename scalar_type,
250  typename decoder_jac_type
251 >
252 ::pressio::mpl::enable_if_t<
253  containers::predicates::is_multi_vector_tpetra<lspg_matrix_type>::value and
254  containers::predicates::is_multi_vector_tpetra<decoder_jac_type>::value
255  >
256 time_discrete_jacobian(lspg_matrix_type & jphi, //jphi holds J * phi
257  const scalar_type & dt,
258  const decoder_jac_type & phi){
259 
260  //get row map of phi
261  const auto phi_map = phi.getMap();
262  // get my global elements
263  const auto gIDphi = phi_map->getMyGlobalIndices();
264 
265  // get map of jphi
266  const auto jphi_map = jphi.getMap();
267  // get global elements
268  const auto gIDjphi = jphi_map->getMyGlobalIndices();
269 
270  // prefactor (f) multiplying f*dt*J*phi
271  constexpr auto prefactor = dtPrefactor<stepper_tag, scalar_type>::value;
272 
273  auto jphi2dView = jphi.get2dViewNonConst();
274  auto phi2dView = phi.get2dView();
275 
276  //loop over elements of jphi
277  for (size_t i=0; i<jphi.getLocalLength(); i++){
278  // ask the phi map what is the local index corresponding
279  // to the global index we are handling
280  const auto lid = phi_map->getLocalElement(gIDjphi[i]);
281  for (size_t j=0; j<jphi.getNumVectors(); j++)
282  jphi2dView[j][i] = phi2dView[j][lid] + prefactor*dt*jphi2dView[j][i];
283  }
284 }
