147  time_discrete_jacobian(const fom_type & fomSystemObj,
148  const fom_state_type & fomState,
149  jac_type & Jphi,
150  const basis_type & phi,
151  const scalar_type & t,
152  const scalar_type & dt,
153  const ::pressio::rom::wls::window_size_t & step,
154  int arg=0 ) const
155  {
156  // u^n - u^{n-1} - f ;
157  if (arg == 0){
158  fomSystemObj.applyJacobian(*fomState.data(), *phi.data(), t, *(Jphi).data());
159  constexpr auto cn = ::pressio::ode::constants::bdf1<scalar_type>::c_n_; // 1
160  const auto cfdt = ::pressio::ode::constants::bdf1<scalar_type>::c_f_*dt; // -1*dt
161  ::pressio::ops::do_update(Jphi, cfdt, phi, cn);
162  }
163 
164  //only perform computation once since this never changes
165  if (arg == 1 && jacobianNeedsRecomputing_){
166  constexpr auto cnm1 = ::pressio::ode::constants::bdf1<scalar_type>::c_nm1_; // -1.
167  ::pressio::ops::do_update(Jphi, phi, cnm1);
168  jacobianNeedsRecomputing_ = true;
169  }
170  }
