74  ::pressio::containers::predicates::is_multi_vector_wrapper_kokkos<MatrixT>::value,
75  "Kokkos direct dense solver expects either (a) dense matrix wrapper or a (b) multi-vector wrapper, both wrapping a rank=2 Kokkos View");

113 #ifdef PRESSIO_ENABLE_TPL_TRILINOS
114  /*
115  * enable if:
116  * the matrix has layout left (i.e. column major)
117  * T is a kokkos vector wrapper
118  * T has host execution space
119  * T and MatrixT have same execution space
120  */
121  template <typename _MatrixT = MatrixT, typename T>
122  mpl::enable_if_t<
123  mpl::is_same<
124  typename ::pressio::containers::details::traits<_MatrixT>::layout, Kokkos::LayoutLeft
125  >::value and
126  ::pressio::containers::predicates::is_vector_wrapper_kokkos<T>::value and
127  ::pressio::containers::details::traits<T>::has_host_execution_space and
128  mpl::is_same<
129  typename containers::details::traits<T>::execution_space,
130  typename containers::details::traits<_MatrixT>::execution_space
131  >::value
132  >
133  solveAllowMatOverwriteImpl(_MatrixT & A, const T& b, T & y)
134  {
135  // gerts is for square matrices
136  assert(A.extent(0) == A.extent(1));
137 
138  assert(A.extent(0) == b.extent(0) );
139  assert(A.extent(1) == y.extent(0) );
140 
141  // only one rhs because this is only enabled if T is a vector wrapper
142  constexpr int nRhs = 1;
143 
144  // just use n, since rows == cols
145  const auto n = A.extent(0);
146 
147  // to store the return code of the function
148  int info = 0;
149 
150  if (tau_.size() != n)
151  tau_.resize(n);
152 
153  if (lwork_ == -1){
154  // this means we need to query what lwork should be
155  // and resize the work array "work_" properly
156  lpk_.GEQRF(n, n, A.data()->data(), n, tau_.data(), work_.data(), lwork_, &info);
157  work_.resize(work_[0]);
158  lwork_ = work_[0];
159  }
160  // do QR
161  lpk_.GEQRF(n, n, A.data()->data(), n, tau_.data(), work_.data(), lwork_, &info);
162  assert(info == 0);
163  // std::cout << " info " << info
164  // << " lwork " << lwork_
165  // << " workSz " << work_.size() << std::endl;
166 
167  // Note that A is overwritten starting from here!
168 
169  // we need to deep copy b into y and pass y to ormqr
170  // because it is overwritten with Q^T b
171  Kokkos::deep_copy(*y.data(), *b.data());
172 
173  // compute Q^T b
174  const char side = 'L';
175  const char trans = 'T';
176  lpk_.ORMQR(side, trans, n, nRhs, n,
177  A.data()->data(),
178  n,
179  tau_.data(),
180  y.data()->data(),
181  n,
182  work_.data(),
183  lwork_,
184  &info);
185  //std::cout << " info-ormqr " << info << std::endl;
186 
187  // solver R y = Q^T b
188  constexpr scalar_t alpha = ::pressio::utils::constants<scalar_t>::one();
189  blas_.TRSM(Teuchos::ESide::LEFT_SIDE,
190  Teuchos::EUplo::UPPER_TRI,
191  Teuchos::ETransp::NO_TRANS,
192  Teuchos::EDiag::NON_UNIT_DIAG,
193  n, nRhs, alpha,
194  A.data()->data(), n, y.data()->data(), n);
195  }
196 #endif
197 
198 
199 private:
200 
201 #ifdef PRESSIO_ENABLE_TPL_TRILINOS
202  Teuchos::LAPACK<int, scalar_t> lpk_;
203  Teuchos::BLAS<int, scalar_t> blas_;
204 
205  // lwork is used for the geqrf, but if you read the doc,
206  // if lwork == -1, then geqrf does a query of what is needed.
207  // more details are shown in the code above on how we use lwork
208  int lwork_ = -1;
209 
210  std::vector<scalar_t> work_ = {0};
211  std::vector<scalar_t> tau_ = {};
212 #endif
213 
214 #if defined PRESSIO_ENABLE_TPL_KOKKOS and defined KOKKOS_ENABLE_CUDA
215  cusolverDnHandle_t cuDnHandle_;
216 #endif
217 };
