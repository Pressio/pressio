64 template<typename MatrixT>
65 class KokkosDirect<::pressio::solvers::linear::direct::potrsL, MatrixT>
66  : public LinearBase<MatrixT, KokkosDirect<::pressio::solvers::linear::direct::potrsL, MatrixT>>
67 {
68 public:
69  static_assert( ::pressio::containers::predicates::is_dense_matrix_wrapper_kokkos<MatrixT>::value or
70  ::pressio::containers::predicates::is_multi_vector_wrapper_kokkos<MatrixT>::value,
71  "Kokkos direct dense solver expects either (a) dense matrix wrapper or a (b) multi-vector wrapper, both wrapping a rank=2 Kokkos View");
72 
73  using solver_tag = ::pressio::solvers::linear::direct::potrsL;
74  using solver_traits = linear::details::traits<solver_tag>;
75  using this_t = KokkosDirect<solver_tag, MatrixT>;
76  using matrix_type = MatrixT;
77  using native_mat_t = typename containers::details::traits<MatrixT>::wrapped_t;
78  using scalar_t = typename containers::details::traits<MatrixT>::scalar_t;
79  using exe_space = typename containers::details::traits<MatrixT>::execution_space;
80  using base_t = LinearBase<MatrixT, this_t>;
81  friend base_t;
82 
83  static_assert( solver_traits::kokkos_enabled == true,
84  "the native solver must suppport kokkos to use in KokkosDirect");
85  static_assert( solver_traits::direct == true,
86  "the native solver must be direct to use in KokkosDirect");
87 
88 
89 public:
90  KokkosDirect() = default;
91  KokkosDirect(const KokkosDirect &) = delete;
92  ~KokkosDirect() = default;
93 
94 private:
95 
96 // because this uses teuchos lapack wrapper
97 #ifdef PRESSIO_ENABLE_TPL_TRILINOS
98  /*
99  * enable if:
100  * the matrix has layout left (i.e. column major)
101  * T is a kokkos vector wrapper
102  * has host execution space
103  * T and MatrixT have same execution space
104  */
105  template < typename _MatrixT = MatrixT, typename T>
106  mpl::enable_if_t<
107  mpl::is_same<
108  typename ::pressio::containers::details::traits<_MatrixT>::layout, Kokkos::LayoutLeft
109  >::value and
110  ::pressio::containers::predicates::is_vector_wrapper_kokkos<T>::value and
111  ::pressio::containers::details::traits<T>::has_host_execution_space and
112  mpl::is_same<
113  typename containers::details::traits<T>::execution_space,
114  typename containers::details::traits<_MatrixT>::execution_space
115  >::value
116  >
117  solveAllowMatOverwriteImpl(_MatrixT & A, const T& b, T & y) {
118  assert(A.extent(0) == b.extent(0) );
119  assert(A.extent(1) == y.extent(0) );
120  // potrs is for symmetric pos def
121  assert(A.extent(0) == A.extent(1) );
122 
123  // only one rhs because this is only enabled if T is a vector wrapper
124  constexpr int nRhs = 1;
125 
126  // just use n, since rows == cols
127  const auto n = A.extent(0);
128 
129  // Cholesky factorization
130  int info = 0;
131  lpk_.POTRF(uplo_, n, A.data()->data(), n, &info);
132  assert(info == 0);
133 
134  // we need to deep copy b into y and pass y
135  // because we overwrite the RHS in place with the solution
136  Kokkos::deep_copy(*y.data(), *b.data());
137 
138  lpk_.POTRS(uplo_, n, nRhs, A.data()->data(), n,
139  y.data()->data(), y.extent(0), &info);
140  assert(info == 0);
141  }
142 #endif
143 
144  const char uplo_ = 'L';
145 
146 #ifdef PRESSIO_ENABLE_TPL_TRILINOS
147  Teuchos::LAPACK<int, scalar_t> lpk_;
148 #endif
149 };
