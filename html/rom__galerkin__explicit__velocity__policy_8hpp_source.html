105  galerkin_state_t create(const fom_t & app) const

113  /* for now, the ROM state and ROM velocity must be of the same type */
114  template <typename fom_t, typename scalar_t>
115  void compute(const galerkin_state_t & romState,
116  galerkin_state_t & romRhs,
117  const fom_t & app,

123 private:
124  //--------------------------------------------
125  // query fom velocity
126  //--------------------------------------------
127  template<
128  typename scalar_t,
129  typename fom_system_t,
130  typename fom_state_t,
131  typename _fom_rhs_t = fom_rhs_t
132  >
133  ::pressio::mpl::enable_if_t<
134  ::pressio::containers::predicates::is_wrapper<fom_state_t>::value
135 #ifdef PRESSIO_ENABLE_TPL_PYBIND11
136  and ::pressio::mpl::not_same<fom_system_t, pybind11::object>::value
137 #endif
138  >
139  queryFomVelocity(const fom_system_t & fomSystemObj,
140  const fom_state_t & fomState,
141  const scalar_t & time) const
142  {
143  fomSystemObj.velocity(*fomState.data(), time, *fomRhs_.data());
144  }

146 #ifdef PRESSIO_ENABLE_TPL_PYBIND11
147  template<

164  // --------------------------------------------
165  // compute RHS of ode
166  // --------------------------------------------
167  template <
168  typename scalar_t,
169  typename result_t,
170  typename _ops_t = ud_ops
171  >
172  ::pressio::mpl::enable_if_t< std::is_void<_ops_t>::value >
173  applyDecoderJacobianToFomVel(result_t & result) const
174  {
175  constexpr auto zero = ::pressio::utils::constants<scalar_t>::zero();
176  constexpr auto one = ::pressio::utils::constants<scalar_t>::one();
177  ::pressio::ops::product(::pressio::transpose(), one, phi_, fomRhs_, zero, result);
178  }
179 
180  template <
181  typename scalar_t,
182  typename result_t,
183  typename _ops_t = ud_ops
184  >
185  ::pressio::mpl::enable_if_t< !std::is_void<_ops_t>::value >
186  applyDecoderJacobianToFomVel(result_t & result) const
187  {
188  constexpr auto zero = ::pressio::utils::constants<scalar_t>::zero();
189  constexpr auto one = ::pressio::utils::constants<scalar_t>::one();
190  udOps_->product(::pressio::transpose(), one, *phi_.data(), *fomRhs_.data(), zero, result);
191  }

193  template <typename fom_system_t, typename scalar_t>
194  void compute_impl(const galerkin_state_t & romState,
195  galerkin_state_t & romRhs,
196  const fom_system_t & fomSystemObj,
197  const scalar_t & t) const
198  {
199 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
200  auto timer = Teuchos::TimeMonitor::getStackedTimer();
201  timer->start("galerkin explicit velocity");
202 #endif
203 
204  fomStatesMngr_.reconstructCurrentFomState(romState);
205 
206 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
207  timer->start("fom eval rhs");
208 #endif
209  const auto & yFom = fomStatesMngr_.getCRefToCurrentFomState();
210  (*this).template queryFomVelocity<scalar_t>(fomSystemObj, yFom, t);
211 
212 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
213  timer->stop("fom eval rhs");
214  timer->start("phiT*fomRhs");
215 #endif
216 
217  (*this).template applyDecoderJacobianToFomVel<scalar_t>(romRhs);
218 
219 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
220  timer->stop("phiT*fomRhs");
221  timer->stop("galerkin explicit velocity");
222 #endif
223  }
224 
225 protected:

228  fom_states_manager_t & fomStatesMngr_;
229  const ud_ops * udOps_ = {};
