200 /* when we have hyper-reduction and we need to calculate the
201  * time-discrete residual, yn (i.e. the current state) and
202  * prevStates (i.e. the states at prev steps) might have
203  * different entries/maps than R (i.e. the spatial residual)
204  * so we need to update R only picking elements corresponding
205  * elements in yn and prevStates.
206  * In other words, in the functions below we do NOT assume that
207  * yn and prevStates have the same distribution/sizes of R but we DO assume
208  * that yn and prevStates contain at least all the elements of R (and more).
209  * THis is because to evaluate spatial residual one typically
210  * needs neighbors values, so in the hyper-red scenario the
211  * states contain more DOFs than the spatial residual.
212  * for instance, imagine the i-th cell in a uniform grid with a
213  * centered 2nd-order FD operator.
214  * To compute the spatial residual at the i-th cell, one needs the
215  * cell value y_i, as well as the left, y_i-1, and right, y_i+1,
216  * values of the state. So for a single DOF of the residual, one
217  * needs 3 DOFs of the state. Extrapolating to all cells, then we
218  * clearly see that we need more dofs for the states than for the residual.
219  *
220  * Below, we account for this potential scenario and the code below
221  * obviously also works for the case where the vectors have same
222  * the same distributions.
223  */

271 template<
272  typename stepper_tag,
273  typename fom_states_manager_t,
274  typename state_type,
275  typename scalar_type
276  >
277 ::pressio::mpl::enable_if_t<
278  containers::predicates::is_vector_wrapper_epetra<state_type>::value == true
279  >
280 time_discrete_residual(const fom_states_manager_t & fomStatesMngr,
281  state_type & R,
282  scalar_type dt){
283  // On input: R contains the application RHS, i.e. if
284  // dudt = f(x,u,...), R contains f(...)
285 
286  const auto & y_n = fomStatesMngr.getCRefToCurrentFomState();
287 
288  // the integral type of the global indices
289  using GO_t = typename containers::details::traits<state_type>::global_ordinal_t;
290 
291  // get map of y_n (prevStates has for sure the same map as y_n)
292  const auto & y_map = y_n.data()->Map();
293  // get my global elements
294  std::vector<GO_t> gIDy( y_n.extentLocal(0) );
295  y_map.MyGlobalElements( gIDy.data() );
296 
297  // get map of R
298  const auto & R_map = R.data()->Map();
299  // get global elements
300  std::vector<GO_t> gIDr( R.extentLocal(0) );
301  R_map.MyGlobalElements( gIDr.data() );
302 
303  //loop over elements of R
304  for (auto i=0; i<R.extentLocal(0); i++){
305  // ask the state map what is the local index corresponding
306  // to the global index we are handling
307  const auto lid = y_map.LID(gIDr[i]);
308  // compute the time-discrete entry
309  time_discrete_single_entry_epetra<
310  stepper_tag
311  >::template evaluate<
312  scalar_type, fom_states_manager_t>(dt, R[i], lid, fomStatesMngr);
313  }
314 }
