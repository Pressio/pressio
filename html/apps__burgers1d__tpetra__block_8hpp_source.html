104  void velocity(const state_type & u,
105  const scalar_type t,
106  velocity_type & rhs) const
107  {
108 
109  double valueFromLeft = 0.0;
110  constexpr int tag_ = 1;
111 
112  if( myRank_ < totRanks_ - 1 ){
113  auto uC = u.getLocalBlock(NumMyElem_ - 1);
114  MPI_Send( &uC(0), 1, MPI_DOUBLE,
115  myRank_+1, tag_, *comm_->getRawMpiComm() );
116  }
117  if( myRank_ > 0 ){
118  MPI_Status status;
119  MPI_Recv(&valueFromLeft, 1, MPI_DOUBLE,
120  myRank_-1, tag_,
121  *comm_->getRawMpiComm(), &status); }
122  lo_t i=0;
123  scalar_type uim1;
124  scalar_type rhsLoc;
125  for (auto const & it : myGel_)
126  {
127  uim1 = valueFromLeft;
128  const auto uCi = u.getLocalBlock(i)(0);
129  if (it==0)
130  uim1 = mu_[0];
131  if (i>0)
132  uim1 = u.getLocalBlock(i-1)(0);
133 
134  rhsLoc = ( 0.5*(uim1*uim1 - uCi*uCi) )/dx_;
135  rhs.replaceLocalValues(i,&rhsLoc);
136  i++;
137  }
138 
139  auto xgrid_v = xGrid_->getDataNonConst();
140  for (i=0; i<NumMyElem_; ++i){
141  auto rhsVal = rhs.getLocalBlock(i);
142  rhsLoc = rhsVal(0) + mu_[1]*exp(mu_[2] * xgrid_v[i]);
143  rhs.replaceLocalValues(i,&rhsLoc);
144  }
145  }

225  void computeJacobianUpdate(const state_type & u,
226  jacobian_type & jac,
227  const scalar_type ) const
228  {
229  const scalar_type buffer = exchangeFlux(u);
230 
231  const lo_t* localColInds;
232  scalar_type* vals;
233  for (lo_t i=0; i<NumMyElem_; i++)
234  {
235  auto thisGID = myGel_[i];
236  auto uC = u.getLocalBlock(i);
237  if (thisGID==0)
238  {
239  lo_t numEntries = 1;
240  int err = 0;
241  err = jac.getLocalRowView(i, localColInds, vals, numEntries);
242  if (err != 0) {
243  break;
244  }
245  vals[0] = -dxInv_ * uC(0);
246  }
247  else
248  {
249  lo_t numEntries = 2;
250  int err = 0;
251  err = jac.getLocalRowView(i, localColInds, vals, numEntries);
252  (void)err;
253 
254  if (i==0){
255  // don't know why the indices here must be inverted, but this is right/
256  // looks like block does strange here for storing the col indices of this point
257  vals[1] = dxInv_ * buffer;
258  vals[0] = -dxInv_ * uC(0);
259  }
260  if (i>0){
261  vals[0] = dxInv_ * u.getLocalBlock(i-1)(0);
262  vals[1] = -dxInv_ * uC(0);
263  }
264  }
265  }
266  }//end

268  void assembleGraph(crs_graph_type & graph)
269  {
270  // using tarr_dt = Teuchos::ArrayView<scalar_type>;
271  using tarr_it = Teuchos::ArrayView<go_t>;
272  std::array<go_t,1> ci1;
273  std::array<go_t,2> ci2;
274  for (lo_t i=0; i<NumMyElem_; i++){
275  auto thisGID = myGel_[i];
276  if (thisGID==0){
277  ci1[0] = 0;
278  graph.insertGlobalIndices(thisGID, tarr_it(ci1.data(),1));
279  }
280  else{
281  ci2[0] = thisGID-1;
282  ci2[1] = thisGID;
283  graph.insertGlobalIndices(thisGID, tarr_it(ci2.data(),2));
284  }
285  }
286  graph.fillComplete();
287  }//end

289  scalar_type exchangeFlux(const state_type & u) const
290  {
291  // to populate the jacobian each process needs the last grid
292  // point solution from the previous process
293  scalar_type buffer {0.0};
294  MPI_Status st;
295  constexpr int tag_ = 1;
296  if (myRank_ < totRanks_-1){
297  auto uC = u.getLocalBlock(NumMyElem_ - 1);
298  MPI_Send(&uC(0), 1, MPI_DOUBLE,
299  myRank_+1, tag_, *comm_->getRawMpiComm());
300  }
301  if (myRank_ >= 1){
302  MPI_Recv(&buffer, 1, MPI_DOUBLE, myRank_-1, tag_,
303  *comm_->getRawMpiComm(), &st);
304  }
305  return buffer;
306  }
