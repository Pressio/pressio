52 /*
53 This header file contains the class used for computing the hessian and gradients in the WLS system.
54 The hessian_gradient policy is responsible for assembling Jw^T Jw and J^T rw,
55 where Jw is the reduced windowed Jacobian and rw the residual.
56 The policy initializes in memory:
57 wlsJacs: this is a vector container of n_s -1 local Jacobians, J, where n_s is the width of the time stencil (e.g., n_s = 3 for BDF2)
58 residual: this is the residual vector for the FOM
59 yFOM_current: this is a working variable for the fom state.
60 */

149  template <
150  typename wls_state_type,
151  typename fom_state_reconstr_t,
152  typename hess_type,
153  typename gradient_type
154  >
155  void operator()(const wls_state_type & wlsState,
156  const wls_state_type & wlsStateIC,
157  hess_type & hess,
158  gradient_type & gradient,
159  const fom_state_reconstr_t & fomStateReconstrObj,
160  const scalar_t dt,
161  const window_size_t numStepsInWindow,
162  const scalar_t ts ,
163  const window_size_t step_s,
164  scalar_t & rnorm) const
165  {
166  constexpr auto zero = ::pressio::utils::constants<scalar_t>::zero();
167  constexpr auto one = ::pressio::utils::constants<scalar_t>::one();
168 
169  window_size_t stepNumLocal = 0;
170  scalar_t t = ts + stepNumLocal*dt;
171  window_size_t stepNumGlobal = step_s + stepNumLocal;
172  ::pressio::ops::set_zero(hess);
173  ::pressio::ops::set_zero(gradient);
174 
175  //get access to the state at the first window
176  setCurrentFomState(wlsState, 0, fomStateReconstrObj);
177 
178  //reconstruct the FOM states from the previous window/ICs
179  timeSchemeObj_.updateStatesFirstStep(wlsStateIC, fomStateReconstrObj);
180 
181  //compute the time discrete residual
182 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
183  auto timer = Teuchos::TimeMonitor::getStackedTimer();
184  timer->start("residual");
185 #endif
186  timeSchemeObj_.time_discrete_residual(fomSystemObj_,fomStateCurrent_, residual_, ts, dt, stepNumGlobal);
187 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
188  timer->stop("residual");
189 #endif
190  Preconditioner(fomSystemObj_, fomStateCurrent_, residual_, t);
191 
192  //increment the norm
193  rNormHelper_ = ::pressio::ops::norm2(residual_);
194  rnorm += rNormHelper_*rNormHelper_;
195 
196  // compute jacobian over stencil
197  if (innerLoopCounter_ % jacobianUpdateFrequency_ == 0)
198  {
199 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
200  timer->start("jacobian");
201 #endif
202  computeJacobiansOverStencil(timeSchemeObj_, stepNumGlobal, stepNumLocal , t,dt);
203 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
204  timer->stop("jacobian");
205 #endif
206  } //end if for freezing
207 
208  // add to local block of hessian
209 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
210  timer->start("hessian computation");
211 #endif
212  auto hess_block = ::pressio::containers::subspan(hess,
213  std::make_pair( stepNumLocal*romSize_,(stepNumLocal+1)*romSize_ ) ,
214  std::make_pair( stepNumLocal*romSize_,(stepNumLocal+1)*romSize_) );
215  ::pressio::ops::product(::pressio::transpose(), ::pressio::nontranspose(),
216  one, jacobians_.getLocalJacobian(stepNumLocal,0),
217  jacobians_.getLocalJacobian(stepNumLocal,0), zero, hess_block);
218 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
219  timer->stop("hessian computation");
220 #endif
221 
222  // compute gradient[n*romSize_:(n+1)*romSize] += J^T r
223 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
224  timer->start("gradient computation");
225 #endif
226  auto gradientView = ::pressio::containers::span(gradient, stepNumLocal*romSize_, romSize_);
227  ::pressio::ops::product(::pressio::transpose(),
228  one, jacobians_.getLocalJacobian(stepNumLocal,0), residual_,
229  one, gradientView);
230 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
231  timer->stop("gradient computation");
232 #endif
233  for (window_size_t stepNumLocal = 1; stepNumLocal < numStepsInWindow; stepNumLocal++)
234  {
235  updateResidualAndJacobian(timeSchemeObj_, wlsState, fomStateReconstrObj, stepNumLocal,
236  step_s, ts, rnorm, gradient, dt);
237 
238  const window_size_t sbar = std::min(stepNumLocal, jacStencilSize_);
239  for (window_size_t i=0; i < sbar; i++)
240  {
241 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
242  timer->start("gradient computation");
243 #endif
244  auto gradientView = ::pressio::containers::span(gradient, (stepNumLocal-i)*romSize_, romSize_);
245  ::pressio::ops::product(::pressio::transpose(), one,
246  jacobians_.getLocalJacobian( stepNumLocal, i ), residual_, one, gradientView);
247 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
248  timer->stop("gradient computation");
249 #endif
250  }
251 
252 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
253  timer->start("hessian computation");
254 #endif
255  addToHessian(hess, stepNumLocal, sbar, hessian_matrix_structure_tag());
256 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
257  timer->stop("hessian computation");
258 #endif
259 
260  }//end loop over stepsInWindow
261  innerLoopCounter_ += 1;
262  rnorm = std::sqrt(rnorm);
263  }//end operator()

335  void addToHessian(hessian_type & hess,
336  const window_size_t & n,
337  const window_size_t & sbar,
338  const ::pressio::matrixUpperTriangular & hessianTag) const
339  {
340  for (window_size_t i=0; i < sbar; i++){

342  constexpr auto one = ::pressio::utils::constants<scalar_t>::one();
343  auto hess_block = ::pressio::containers::subspan(hess,
344  std::make_pair( (n-i)*romSize_, (n-i+1)*romSize_ ),
345  std::make_pair( (n-j)*romSize_,(n-j+1)*romSize_ ) );
346  ::pressio::ops::product(::pressio::transpose(), ::pressio::nontranspose(), one,
347  jacobians_.getLocalJacobian( n, i ),
348  jacobians_.getLocalJacobian( n, j ), one, hess_block);
349  }

439  rnorm += rNormHelper_*rNormHelper_;

441 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
