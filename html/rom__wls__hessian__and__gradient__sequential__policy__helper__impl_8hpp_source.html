66  using jacUpdatingTagIC = ::pressio::mpl::variadic::find_if_unary_pred_t<
67  ::pressio::rom::wls::predicates::is_legitimate_jacobian_updating_tag, Args...>;
68  using jac_update_t = ::pressio::mpl::variadic::at_or_t<::pressio::rom::wls::NonFrozenJacobian,

70 
71  // set the container type based on the tag
72  using jac_t = typename decoder_type::jacobian_type;
73  using jac_cont_t = typename std::conditional<
74  std::is_same<jac_update_t, ::pressio::rom::wls::NonFrozenJacobian>::value,
75  ::pressio::rom::wls::NonFrozenJacobiansContainer<jac_t>,
76  ::pressio::rom::wls::FrozenJacobiansContainer<jac_t>
77  >::type;
78 
79  // find if Args contain a valid precond type, if not default = NoPreconditioner
80  using precIC = ::pressio::mpl::variadic::find_if_unary_pred_t<::pressio::rom::wls::predicates::is_legitimate_preconditioner_type, Args...>;
81  using prec_t = ::pressio::mpl::variadic::at_or_t<::pressio::rom::wls::preconditioners::NoPreconditioner, precIC::value, Args...>;
82 
83  // final type
84  using type = ::pressio::rom::wls::impl::HessianGradientSequentialPolicy<
85  fom_system_type, decoder_type, ode_tag, hess_structure_tag, prec_t, jac_cont_t>;
