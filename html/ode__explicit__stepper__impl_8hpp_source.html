102 // Args can contain : ud_ops, velocity_policy, velocity_type in any order
103 
104 template<typename tag, typename state_type, typename system_t, typename ...Args>
105 struct Stepper
106 {
107  using state_t = state_type;
108 
109  // // check if scalar is provided in Args
110  // using ic0 = ::pressio::mpl::variadic::find_if_unary_pred_t<
111  // std::is_floating_point, Args...>;
112  // using scalar_t = ::pressio::mpl::variadic::at_or_t<
113  // void, ic0::value, Args...>;
114  // static_assert(std::is_void<scalar_t>::value == false,
115  // "You need a scalar_type in the ExplicitStepper templates");
116  using scalar_t = typename ::pressio::containers::details::traits<state_type>::scalar_t;
117 
118  static_assert(::pressio::ode::concepts::continuous_time_explicit_system<system_t>::value,
119  "Invalid system passed to the ExplicitStepper");
120 
121  // check args for a valid velocity type
122  using ic2 = ::pressio::mpl::variadic::find_if_unary_pred_t<
123  ::pressio::ode::concepts::explicit_velocity, Args...>;
124  // if a velocity type is NOT found, then set it equal to the state
125  using velocity_t = ::pressio::mpl::variadic::at_or_t<state_t, ic2::value, Args...>;
126  static_assert(std::is_void<system_t>::value == false,
127  "The velicity type cannot be void");
128 
129  // this is the standard velocity policy (just typedef, it is only used
130  // if the user does not pass a user-defined policy)
131  using standard_velocity_policy_t = ::pressio::ode::explicitmethods::policy::VelocityStandardPolicy<
132  state_t, system_t, velocity_t>;
133 
134  // check Args if a user-defined velocity policy is passed
135  using ic3 = ::pressio::mpl::variadic::find_if_quinary_pred_t<
136  scalar_t, state_type, velocity_t, system_t,
137  ::pressio::ode::concepts::explicit_velocity_policy, Args...>;
138  using velocity_policy_t = ::pressio::mpl::variadic::at_or_t<standard_velocity_policy_t, ic3::value, Args...>;
139 
140  // check for user-defined ops
141  using ops_t = typename UserDefinedOpsFilter<tag, scalar_t, state_t, velocity_t, Args...>::type;
142 
143  // implementation class type
144  using type = typename ImplSelector<tag>::template type<scalar_t, state_t, system_t, velocity_t,
145  velocity_policy_t, standard_velocity_policy_t, ops_t>;
146 };
