54 template<typename derived_type>
55 class VectorSharedMemBase
56  : public ContainerSharedMemBase<derived_type>
57 {
58  static_assert(details::traits<derived_type>::is_shared_mem==1,
59  "OOPS: distributed concrete vector inheriting from sharedMem base!");
60 
61  using mytraits = typename details::traits<derived_type>;
62  using ord_t = typename mytraits::ordinal_t;
63  using ref_t = typename mytraits::reference_t;
64  using const_ref_t = typename mytraits::const_reference_t;
65 
66 public:
67  bool empty() const{
68  return static_cast<derived_type const *>(this)->empty();
69  }
70 
71  ref_t operator[](ord_t i){
72  return static_cast<derived_type &>(*this)[i];
73  }
74 
75  const_ref_t operator[](ord_t i) const{
76  return static_cast<derived_type const &>(*this)[i];
77  }
78 
79 private:
80  /* workaround for nvcc issue with templates, see https://devtalk.nvidia.com/default/topic/1037721/nvcc-compilation-error-with-template-parameter-as-a-friend-within-a-namespace/ */
81  template<typename DummyType> struct dummy{using type = DummyType;};
82  friend typename dummy<derived_type>::type;
83 };//end class
