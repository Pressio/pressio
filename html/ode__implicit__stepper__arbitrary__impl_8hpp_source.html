157  void doStep(ode_state_type & odeState,
158  const scalar_t & t,
159  const scalar_t & dt,
160  const types::step_t & step,
161  solver_type & solver)

167  this->dt_ = dt;

169  this->step_ = step;

171  constexpr auto nAux = decltype(this->auxStates_)::size();
172  this->updateAuxiliaryStorage<nAux>(odeState);

179  void updateAuxiliaryStorage(const ode_state_type & odeState){
180  // copy y_n into y_n-1

185  // when we have two aux states,
186  template<std::size_t nAux, mpl::enable_if_t<nAux==2, int> = 0>
187  void updateAuxiliaryStorage(const ode_state_type & odeState){
188  // copy y_n-1 into y_n-2
189  auto & y_nm1 = this->auxStates_.get(ode::nMinusOne());
190  auto & y_nm2 = this->auxStates_.get(ode::nMinusTwo());
191  ::pressio::ops::deep_copy(y_nm2, y_nm1);
192  // copy y_n into y_n-1
193  ::pressio::ops::deep_copy(y_nm1, odeState);
194  }

199  auto & y_nm1 = this->auxStates_.get(ode::nMinusOne());
200  auto & y_nm2 = this->auxStates_.get(ode::nMinusTwo());
201  auto & y_nm3 = this->auxStates_.get(ode::nMinusThree());
202  ::pressio::ops::deep_copy(y_nm3, y_nm2);
203  ::pressio::ops::deep_copy(y_nm2, y_nm1);
204  ::pressio::ops::deep_copy(y_nm1, odeState);
205  }

210  auto & y_nm1 = this->auxStates_.get(ode::nMinusOne());
211  auto & y_nm2 = this->auxStates_.get(ode::nMinusTwo());
212  auto & y_nm3 = this->auxStates_.get(ode::nMinusThree());
213  auto & y_nm4 = this->auxStates_.get(ode::nMinusFour());
214  ::pressio::ops::deep_copy(y_nm4, y_nm3);
215  ::pressio::ops::deep_copy(y_nm3, y_nm2);
216  ::pressio::ops::deep_copy(y_nm2, y_nm1);
217  ::pressio::ops::deep_copy(y_nm1, odeState);
218  }
