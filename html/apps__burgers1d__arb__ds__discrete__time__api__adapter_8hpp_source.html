135  // case when we only have a single auxiliary state
136  template <typename step_t, typename state_t>
137  void applyTimeDiscreteJacobianImpl(const step_t & step,
138  const scalar_type & time,
139  const scalar_type & dt,
140  const dense_matrix_type & B,
141  dense_matrix_type & A,
142  const state_t & yn,
143  const state_t & ynm1) const
144  {
145  // compute Jacobian
146  appObj_.jacobian(yn, time, JJ_);
147 
148  // compute time discrete Jacobian
149  for (int_t i=0; i<JJ_.extent(0); ++i){
150  for (int_t j=0; j<JJ_.extent(1); ++j){
151  JJ_(i,j) *= -dt;
152  if (i==j)
153  JJ_(i,j) += ::pressio::utils::constants<scalar_type>::one();
154  }
155  }
156 
157  // compute A = JJ * B
158  for (int_t i=0; i<A.extent(0); ++i){
159  for (int_t j=0; j<A.extent(1); ++j){
160  A(i,j) = ::pressio::utils::constants<scalar_type>::zero();
161  for (int_t k=0; k<JJ_.extent(1); ++k){
162  A(i,j) += JJ_(i,k) * B(k,j);
163  }
164  }
165  }
166  }
