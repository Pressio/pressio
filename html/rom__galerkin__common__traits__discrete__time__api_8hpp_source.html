54 template <
55  typename fom_system_type,
56  typename rom_state_type,
57  typename rom_jacobian_type,
58  typename ...Args
59  >
60 struct CommonTraitsDiscreteTimeApi
61 {
62  // the scalar type
63  using scalar_t = typename ::pressio::containers::details::traits<rom_state_type>::scalar_t;
64 
65  using fom_system_t = fom_system_type;
66  using fom_native_state_t = typename fom_system_t::state_type;
67  using fom_native_residual_t = typename fom_system_t::discrete_time_residual_type;
68 
69  // fom wrapper types
70  using fom_state_t = ::pressio::containers::Vector<fom_native_state_t>;
71  using fom_residual_t = ::pressio::containers::Vector<fom_native_residual_t>;
72 
73  // rom types
74  using rom_state_t = rom_state_type;
75  using rom_residual_t = rom_state_type;
76  using rom_jacobian_t = rom_jacobian_type;
77 
78  // verify that args contains a valid decoder type
79  using ic0 = ::pressio::mpl::variadic::find_if_ternary_pred_t<
80  rom_state_t, fom_state_t, ::pressio::rom::concepts::decoder, Args...>;
81  using decoder_t = ::pressio::mpl::variadic::at_or_t<void, ic0::value, Args...>;
82  static_assert(!std::is_void<decoder_t>::value and ic0::value < sizeof... (Args),
83  "A valid decoder type must be passed");
84  using decoder_jac_t = typename decoder_t::jacobian_type;
85 
86  /* fom_apply_jacobian_t is type of J*decoder_jac_t where
87  * * J is the jacobian of the fom rhs
88  * * decoder_jac_t is the type of the decoder jacobian
89  * For now, set fom_apply_jacobian_t to be of same type as decoder_jac_t
90  * not bad assumption since all matrices are left-applied to decoder_jac_t
91  */
92  using fom_apply_jacobian_t = decoder_jac_t;
93 
94  // if we have an admissible user-defined ops
95  using ud_ops_t = void;
96 
97  // fom state reconstructor type
98  using fom_state_reconstr_t = FomStateReconstructor<scalar_t, fom_state_t, decoder_t>;
99 
100  //-------------------------------
101  // find the order setter in Args
102  //-------------------------------
103  using ic1 = ::pressio::mpl::variadic::find_if_unary_pred_t<
104  ::pressio::ode::predicates::IsStepperOrderSetter, Args...>;
105  using order_setter = ::pressio::mpl::variadic::at_or_t<void, ic1::value, Args...>;
106  static_assert( !std::is_void<order_setter>::value,
107  "To use Galerkin with residual api, you need to set the order of the stepper \n \
108 at compile time by passing to a template argument as follows: \n \
109 ::pressio::ode::types::StepperOrder<your_order_value>.");
110  // store
111  static constexpr ::pressio::ode::types::stepper_order_t order_value = order_setter::value;
112 
113  //-----------------------------------------------------------
114  // find the total number of states needed
115  //-----------------------------------------------------------
116  using ic2 = ::pressio::mpl::variadic::find_if_unary_pred_t<
117  ::pressio::ode::predicates::IsStepperTotalNumStatesSetter, Args...>;
118  using tot_n_setter = ::pressio::mpl::variadic::at_or_t<void, ic2::value, Args...>;
119  static_assert( !std::is_void<tot_n_setter>::value,
120  "\nTo use Galerkin with residual api, you need to set the \
121 total number of states needed for the stepper at compile time by passing \
122 to a template argument as follows: \n \
123 ::pressio::ode::types::StepperTotalNumberOfStates<your_order_value>. \n \
124 Note that this is the total number of states needed including previous ones, \n \
125 basically the size of the stpper stencil.");
126 
127  // total number of fom states needed (size of stencil plus the state at current step)
128  static constexpr std::size_t numStates = tot_n_setter::value;
129 
130  // type of class holding the fom states
131  using fom_states_manager_t = ::pressio::rom::ManagerFomStatesStatic<fom_state_t, numStates, fom_state_reconstr_t, ud_ops_t>;
132 
133 };
