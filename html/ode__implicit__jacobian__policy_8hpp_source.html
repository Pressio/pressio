93  std::declval<system_t const &>(),
94  std::declval<scalar_t const &>(),
95  std::declval<scalar_t const &>(),
96  std::declval<::pressio::ode::types::step_t const &>(),
97  std::declval<jacobian_t &>()
98  )
99  )
100  >::value
101  >
102  > : std::true_type{};
