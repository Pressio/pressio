107  typename stepper_tag,
108  typename lspg_state_t,

111  typename scalar_t

113  void compute(const lspg_state_t & romState,
114  const prev_states_t & romPrevStates,

128  template <
129  typename stepper_tag,
130  typename fom_state_cont_type,
131  typename scalar_t,
132  typename _ud_ops_t = ud_ops_t
133  >
134  ::pressio::mpl::enable_if_t< std::is_void<_ud_ops_t>::value >
135  time_discrete_dispatcher(const fom_state_cont_type & fomStates,
136  residual_t & romR,
137  const scalar_t & dt,
138  ::pressio::Norm normKind,
139  scalar_t & normValue) const
140  {
141  using namespace ::pressio::rom::lspg::impl::unsteady;
142  time_discrete_residual<stepper_tag>(fomStates, romR, dt);
143 
144  if (normKind == ::pressio::Norm::L2)
145  normValue = ::pressio::ops::norm2(romR);
146  else if (normKind == ::pressio::Norm::L1)

152  template <
153  typename stepper_tag,
154  typename fom_state_cont_type,
155  typename scalar_t,
156  typename _ud_ops_t = ud_ops_t
157  >
158  ::pressio::mpl::enable_if_t< !std::is_void<_ud_ops_t>::value >
159  time_discrete_dispatcher(const fom_state_cont_type & fomStates,
160  residual_t & romR,
161  const scalar_t & dt,
162  ::pressio::Norm normKind,
163  scalar_t & normValue) const
164  {
165  using namespace ::pressio::rom::lspg::impl::unsteady;
166  time_discrete_residual<stepper_tag>(fomStates, romR, dt, udOps_);
167 
168  if (normKind == ::pressio::Norm::L2)
169  normValue = udOps_->norm2(*romR.data());
170  else if (normKind == ::pressio::Norm::L1)
171  normValue = udOps_->norm1(*romR.data());
172  else
173  throw std::runtime_error("Invalid norm kind for lspg unsteady residual policy");
174  }

176  template <
177  typename stepper_tag,
178  typename lspg_state_t,
179  typename prev_states_t,
180  typename fom_system_t,
181  typename scalar_t
182  >
183  void compute_impl(const lspg_state_t & romState,
184  residual_t & romR,
185  const prev_states_t & romPrevStates,
186  const fom_system_t & fomSystemObj,
187  const scalar_t & time,
188  const scalar_t & dt,
189  const ::pressio::ode::types::step_t & timeStep,
190  ::pressio::Norm normKind,
191  scalar_t & normValue) const
192  {
193 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
194  auto timer = Teuchos::TimeMonitor::getStackedTimer();
195  timer->start("lspg residual");
196 #endif
197 
198  /* the currrent FOM has to be recomputed every time regardless of
199  * whether the timeStep changes since we might be inside a non-linear solve
200  * where the time step does not change but this residual method
201  * is called multiple times.
202  */
203  fomStatesMngr_.reconstructCurrentFomState(romState);
204 
205  /* the previous FOM states should only be recomputed when time step changes.
206  * no need to reconstruct all the FOM states, we just need to reconstruct
207  * the state at the previous step (i.e. t-dt)
208  */
209  if (storedStep_ != timeStep){
210  fomStatesMngr_ << romPrevStates.get(ode::nMinusOne());
211  storedStep_ = timeStep;
212  }
213 
214 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
215  timer->start("fom eval rhs");
216 #endif
217  ::pressio::rom::queryFomVelocity(fomSystemObj,
218  fomStatesMngr_.getCRefToCurrentFomState(),
219  romR, time);
220 
221 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
222  timer->stop("fom eval rhs");
223  timer->start("time discrete residual");
224 #endif
225 
226  this->time_discrete_dispatcher<stepper_tag>(fomStatesMngr_, romR, dt,
227  normKind, normValue);
228 
229 #ifdef PRESSIO_ENABLE_TEUCHOS_TIMERS
230  timer->stop("time discrete residual");
231  timer->stop("lspg residual");
232 #endif
233  }
234 
235 protected:
236  // storedStep is used to keep track of which step we are doing.
237  // This is used to decide whether we need to update/recompute the previous
238  // FOM states or not. Since it does not make sense to recompute previous
239  // FOM states if we are not in a new time step.
240  mutable ::pressio::ode::types::step_t storedStep_ = {};
241 
242  mutable residual_t R_ = {};
243  fom_states_manager_t & fomStatesMngr_;
244 
245 #ifdef PRESSIO_ENABLE_TPL_PYBIND11
246  // here we do this conditional type because it seems when
247  // ud_ops_t= pybind11::object it only works if we copy the object.
248  // Need to figure out if we can leave ptr in all cases.
249  typename std::conditional<
250  ::pressio::mpl::is_same<ud_ops_t, pybind11::object>::value, ud_ops_t,
251  const ud_ops_t *
252  >::type udOps_ = {};
253 #else
254  const ud_ops_t * udOps_ = {};
255 #endif
256 
257 };//end class
